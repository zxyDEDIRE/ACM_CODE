#include<bits/stdc++.h>
using namespace std;
using ll=long long;
using pii=pair<int,int>;
const int maxn=1e6;
int sa,sb,t;
bool f(int x,int y)
{
    string a;
    if(x<=9)a+='0';
    else a+=(char)('0'+x/10);

    a+=(char)('0'+x%10);

    if(y<=9)a+='0';
    else a+=(char)('0'+y/10);

    a+=(char)('0'+y%10);

    if(a[0]==a[3]&&a[1]==a[2])return 1;
    else return 0;
}
void solve()
{
    scanf("%d:%d",&sa,&sb);
    cin>>t;

    int x=sa,y=sb,flag=0,ans=0;
    while(1)
    {
        if(x==sa&&y==sb&&flag)break;
        flag=1;

        y+=t;
        x+=y/60;
        y=y%60;
        x%=24;
        if(f(x,y))ans++;
    }
    cout<<ans<<endl;
}
int main(){
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);cout.tie(nullptr);
    int __;cin>>__;
    while(__--)
    solve();
    return 0;
}
